% =============================================================================
%                            COMBINATORICS PLUGIN
% =============================================================================
%
% Combinatorial notation for binomial variants, Stirling numbers, and PIE-aware
% falling and rising factorials.
%
% Requires:
%   xamsmath core helpers and mathcommand PIE support.

% =============================================================================
%                          COMBINATORIAL NUMBERS
% =============================================================================

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \multbinom
%
% Purpose:
%   Typesets the multiset coefficient, commonly read as combinations with
%   repetition.
%
% Parameters:
%   #1 - Upper argument, usually the number of available element types.
%   #2 - Lower argument, usually the number of chosen elements.
%
% Usage:
%   \multbinom{n}{k}

\newmathcommand{\multbinom}[2]{\left(\kern-.3em\left(\genfrac{}{}{0pt}{}{#1}{#2}\right)\kern-.3em\right)}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \Stirling
%
% Purpose:
%   Typesets Stirling numbers of the second kind with brace delimiters.
%
% Parameters:
%   #1 - Number of elements being partitioned.
%   #2 - Number of non-empty subsets.
%
% Usage:
%   \Stirling{n}{k}

\newmathcommand{\Stirling}[2]{\left\{\genfrac{}{}{0pt}{}{#1}{#2}\right\}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \stirling
%
% Purpose:
%   Typesets Stirling numbers of the first kind with bracket delimiters.
%
% Parameters:
%   #1 - Number of permuted elements.
%   #2 - Number of cycles.
%
% Usage:
%   \stirling{n}{k}

\newmathcommand{\stirling}[2]{\left[\genfrac{}{}{0pt}{}{#1}{#2}\right]}

% =============================================================================
%                             FACTORIAL SYMBOLS
% =============================================================================

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \fallf
%
% Purpose:
%   Typesets the falling factorial using an underlined exponent. The command is
%   PIE-aware, so subscripts and superscripts following the argument are absorbed
%   into the generated symbol.
%
% Parameters:
%   #1 - Base expression.
%
% PIE parameters:
%   #2 - Prime tokens absorbed by mathcommand.
%   #3 - Index tokens absorbed by mathcommand.
%   #4 - Exponent tokens absorbed by mathcommand.
%
% Usage:
%   \fallf{x}^{n}
%   \fallf{x}_{k}^{n}

\newcommandPIE{\fallf}[1]{{#1^{\underline{\GetExponent{#4}}}_{\GetIndex{#3}}}#2}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \risef
%
% Purpose:
%   Typesets the rising factorial, or Pochhammer symbol, using an overlined
%   exponent. The command is PIE-aware, so subscripts and superscripts following
%   the argument are absorbed into the generated symbol.
%
% Parameters:
%   #1 - Base expression.
%
% PIE parameters:
%   #2 - Prime tokens absorbed by mathcommand.
%   #3 - Index tokens absorbed by mathcommand.
%   #4 - Exponent tokens absorbed by mathcommand.
%
% Usage:
%   \risef{x}^{n}
%   \risef{x}_{k}^{n}

\newcommandPIE{\risef}[1]{{#1^{\overline{\GetExponent{#4}}}_{\GetIndex{#3}}}#2}

% =============================================================================
%                                USAGE NOTES
% =============================================================================
%

%

%   \stirling{n}{k}
%
% Falling and rising factorials:
%   \fallf{x}^{n}
%   \fallf{x}_{k}^{n}
%   \risef{x}^{n}
%   \risef{x}_{k}^{n}
