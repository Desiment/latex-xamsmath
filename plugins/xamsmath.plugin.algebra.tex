% =============================================================================
%                               ALGEBRA PLUGIN
% =============================================================================
%
% Algebraic structures, complex numbers, linear algebra notation, vector
% shorthands, matrix operators, and divisibility relations.
%
% Requires:
%   xamsmath core helpers. The braces option is optional because this plugin
%   provides a fallback for \DeclareBracedMathOperator.

% =============================================================================
%                              FALLBACK SUPPORT
% =============================================================================

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \DeclareBracedMathOperator
%
% Purpose:
%   Provides ordinary AMS operator declarations when the braces module has not
%   already supplied the richer braced-operator implementation.
%
% Parameters:
%   *  - Optional star requests limit-style operator placement.
%   #1 - Operator command to define.
%   #2 - Operator text to display.
%
% Usage:
%   \DeclareBracedMathOperator{\tr}{tr}
%   \DeclareBracedMathOperator*{\argmax}{argmax}

\ProvideDocumentCommand{\DeclareBracedMathOperator}{smm}{
  \IfBooleanTF{#1}{\DeclareMathOperator*{#2}{#3}}{\DeclareMathOperator{#2}{#3}}
}

% =============================================================================
%                              COMPLEX NUMBERS
% =============================================================================

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \conj
%
% Purpose:
%   Typesets the complex conjugate of an expression using an overline.
%
% Parameters:
%   #1 - Complex expression to conjugate.
%
% Usage:
%   \conj{z}

\newmathcommand{\conj}[1]{\overline{#1}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \Re
%
% Purpose:
%   Typesets the real-part operator in upright roman text.
%
% Usage:
%   \Re(z)

\let\Re\undefined
\newmathcommand{\Re}{\mathop{\mathrm{Re}}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \Im
%
% Purpose:
%   Typesets the imaginary-part operator in upright roman text.
%
% Usage:
%   \Im(z)

\let\Im\undefined
\newmathcommand{\Im}{\mathop{\mathrm{Im}}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \abs
%
% Purpose:
%   Declares paired absolute-value delimiters for modulus or magnitude notation.
%
% Parameters:
%   #1 - Expression enclosed in vertical bars.
%
% Usage:
%   \abs{z}
%   \abs[\big]{z+w}

\DeclarePairedDelimiter\abs{\lvert}{\rvert}

% =============================================================================
%                          BASIC ALGEBRAIC STRUCTURES

% -----------------------------------------------------------------------------
% Internal Commands
% -----------------------------------------------------------------------------
%
% \bbstruct
%
% Purpose:
%   Centralizes blackboard-bold rendering for standard algebraic structures.
%
% Parameters:
%   #1 - Structure letter to typeset.
%
% Usage:
%   Internal use by \N, \Z, \Q, \R, \C, and \H.

\newcommand*{\bbstruct}[1]{\mathbb{#1}}

    \renewmathcommand{\H}{\bbstruct{H}}
\else
    \newmathcommand{\H}{\bbstruct{H}}
\fi

% =============================================================================
%                              LINEAR ALGEBRA
% =============================================================================

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \norm
%
% Purpose:
%   Declares vector norm delimiters. An empty argument renders a centered dot as
%   a placeholder norm input.
%
% Parameters:
%   #1 - Vector or expression to norm.
%
% Usage:
%   \norm{v}
%   \norm{}

\DeclarePairedDelimiterX\norm[1]\lVert\rVert{\ifblank{#1}{\:\cdot\:}{#1}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \spr
%
% Purpose:
%   Declares scalar-product delimiters with centered-dot placeholders for empty
%   arguments.
%
% Parameters:
%   #1 - First scalar-product argument.
%   #2 - Second scalar-product argument.
%
% Usage:
%   \spr{u}{v}
%   \spr{}{}

\DeclarePairedDelimiterX\spr[2]{\langle}{\rangle}{\ifblank{#1}{\:\cdot\:}{#1},\ifblank{#2}{\:\cdot\:}{#2}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \cpr
%
% Purpose:
%   Declares cross-product-style bracket delimiters with placeholders for empty
%   arguments.
%
% Parameters:
%   #1 - First cross-product argument.
%   #2 - Second cross-product argument.
%
% Usage:
%   \cpr{u}{v}
%   \cpr{}{}

\DeclarePairedDelimiterX\cpr[2]{\lbrack}{\rbrack}{\ifblank{#1}{\:\cdot\:}{#1},\ifblank{#2}{\:\cdot\:}{#2}}

% =============================================================================
%                             VECTOR SHORTHANDS
% =============================================================================

% -----------------------------------------------------------------------------
% Dependencies
% -----------------------------------------------------------------------------
%
% esvect supplies \vv. xamsmath makes \vec use the esvect arrow and then frees
% \vv so package-level vector shorthands can use the v-prefix.

\RequirePackage[d]{esvect}
\RenewCommandCopy{\vec}{\vv}
\let\vv\relax

% -----------------------------------------------------------------------------
% Generated User Commands
% -----------------------------------------------------------------------------
%
% \va through \vz, except letters omitted from the source alphabet list.
%
% Purpose:
%   Generates vector shorthand commands such as \va for \vec{a} and \vx for
%   \vec{x}.
%
% Usage:
%   \va + \vb

\LoopCommands{abcdefghklmnopqrstuvwxyz}[v#1]{\newmathcommand#2{\vec{#1}}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \vi
%
% Purpose:

\newmathcommand{\vi}{\vec{\imath}}

% \vj
%
% Purpose:
%   Typesets the j unit vector with a dotless j under the vector arrow.
%
% Usage:

\newmathcommand{\vj}{\vec{\jmath}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \divby
%
% Purpose:
%   Typesets the Russian divisibility relation using vertically stacked dots.
%
% Usage:
%   a \divby b

\DeclareRobustCommand{\divby}{\mathrel{\text{\vbox{\baselineskip.65ex\lineskiplimit0pt\hbox{.}\hbox{.}\hbox{.}}}}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \bydiv
%
% Purpose:
%   Typesets the international divisibility relation using a vertical bar.
%
% Usage:
%   a \bydiv b

\DeclareRobustCommand{\bydiv}{\mathrel{\mid}}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \notdivby
%
% Purpose:
%   Typesets negated Russian divisibility.
%
% Usage:
%   a \notdivby b

\DeclareRobustCommand{\notdivby}{\centernot\divby}

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \notbydiv
%
% Purpose:
%   Typesets negated international divisibility.
%
% Usage:
%   a \notbydiv b

\DeclareRobustCommand{\notbydiv}{\centernot\bydiv}
