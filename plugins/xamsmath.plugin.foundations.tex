% =============================================================================
%                             FOUNDATIONS PLUGIN
% =============================================================================
%
% Logic, set theory, mappings, and topology notation.
%
% Requires:
%   xamsmath core helpers. The braces option is optional because this plugin
%   provides a fallback for \DeclareBracedMathOperator.

% =============================================================================
%                              FALLBACK SUPPORT
% =============================================================================

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \DeclareBracedMathOperator
%
% Purpose:
%   Provides ordinary AMS operator declarations when the braces module has not
%   already supplied the richer braced-operator implementation.
%
% Parameters:
%   *  - Optional star requests limit-style operator placement.
%   #1 - Operator command to define.
%   #2 - Operator text to display.
%
% Usage:
%   \DeclareBracedMathOperator{\card}{card}

\ProvideDocumentCommand{\DeclareBracedMathOperator}{smm}{
  \IfBooleanTF{#1}{\DeclareMathOperator*{#2}{#3}}{\DeclareMathOperator{#2}{#3}}
}

% =============================================================================
%                                  LOGIC

% -----------------------------------------------------------------------------
% Logical Connectives
% -----------------------------------------------------------------------------
%
% Standard LaTeX logical connective names are relaxed before redefining them as
% mathcommand-aware aliases.

\let\lor\relax
\let\lnot\relax
\let\land\relax

% User Commands
% -----------------------------------------------------------------------------
%
% \lor
%
% Purpose:
%   Typesets logical disjunction.
%
% Usage:
%   p \lor q

% User Commands

% -----------------------------------------------------------------------------
% User Commands
% -----------------------------------------------------------------------------
%
% \set
%
% Purpose:
%   Declares split set-builder delimiters. A single vertical bar separates the
%   set expression from its condition.
%
% Parameters:
%   #1 - Set contents, optionally containing one | separator.
%
% Usage:
%   \set{x | x > 0}
%   \set[\Big]{x | x^2 > 1}
