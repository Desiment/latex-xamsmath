% combinatorics.tex - xamsmath Combinatorics Plugin
% Combinatorial notations: binomial coefficients, Stirling numbers, falling/rising factorials
% Requires xamsmath package


%%%%
%% Combinatorics
%%%%

% Multiset coefficient: ((n k)) - number of ways to choose k elements from n with repetition
\newmathcommand{\multbinom}[2]{\left(\kern-.3em\left(\genfrac{}{}{0pt}{}{#1}{#2}\right)\kern-.3em\right)}

% Stirling numbers of the second kind: {n k} - number of ways to partition n elements into k non-empty subsets
\newmathcommand{\Stirling}[2]{\left\{\genfrac{}{}{0pt}{}{#1}{#2}\right\}}

% Stirling numbers of the first kind: [n k] - number of permutations of n elements with k cycles
\newmathcommand{\stirling}[2]{\left[\genfrac{}{}{0pt}{}{#1}{#2}\right]}

% Falling factorial: x^\underline{n} = x(x-1)(x-2)...(x-n+1)
\newcommandPIE{\fallf}[1]{{#1^{\underline{\GetExponent{#4}}}_{\GetIndex{#3}}}#2}

% Rising factorial (Pochhammer symbol): x^\overline{n} = x(x+1)(x+2)...(x+n-1)
\newcommandPIE{\risef}[1]{{#1^{\overline{\GetExponent{#4}}}_{\GetIndex{#3}}}#2}


% Example usage:
%
% Multiset coefficients:
%   $\multbinom{n}{k}$              % Number of k-combinations with repetition from n elements
%
% Stirling numbers:
%   $\Stirling{n}{k}$               % Stirling numbers of the second kind
%   $\stirling{n}{k}$               % Stirling numbers of the first kind
%
% Falling factorials (note PIE notation):
%   $\fallf{x}^{n}$                  % x^\underline{n} - falling factorial
%   $\fallf{x}_{k}^{n}$             % x_{k}^\underline{n}
%
% Rising factorials (note PIE notation):
%   $\risef{x}^{n}$                  % x^\overline{n} - rising factorial (Pochhammer symbol)
%   $\risef{x}^{n}_{k}$              % x_k^\overline{n}
%
% Combinatorial identities:
%   $\sum_{k=0}^n \Stirling{n}{k} \risef{x}{k} = x^n$  % Expansion of x^n in terms of rising factorials
%   $\fallf{x}{n} = \sum_{k=0}^n \stirling{n}{k} x^k$  % Expansion of falling factorial
